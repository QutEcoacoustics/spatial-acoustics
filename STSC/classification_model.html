labelled_complete <- dplyr::filter(labelled, classID != "") %>% 
  droplevels(.$classID) %>%
  droplevels(.$index)

df_component <- select(labelled_complete, -classID) %>% 
  droplevels(.)

df_class <- select(labelled_complete, -component) %>% 
  droplevels(.)

train_index <- sample(1:nrow(labelled_complete), 0.8 * nrow(labelled_complete))
test_index <- setdiff(1:nrow(labelled_complete), train_index)


train <- labelled_complete[train_index,]%>% 
  droplevels(.$component) %>% 
  droplevels(.$index)


test <- labelled_complete[test_index,]%>% 
  droplevels(.$component) %>% 
  droplevels(.$index)
